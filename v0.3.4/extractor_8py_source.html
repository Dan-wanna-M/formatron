56class LiteralExtractor(Extractor):
57 """
58 An extractor that extracts a literal string.
59 """
60
61 def __init__(self, literal: str):
62 """
63 Initialize the literal extractor. It never captures since capturing a literal is redundant.
64
65 :param literal: The literal string to extract.
66 """
67 super().__init__(None)
68 self._literal = literal

70 def extract(self, input_str: str) -> typing.Optional[tuple[str, typing.Any]]:
71 """
72 Extract the literal from the input string.
73 """
74 pos = input_str.find(self._literal)

104 def extract(self, input_str: str) -> typing.Optional[tuple[str, re.Match | None]]:
105 """
106 Extract the string using the regular expression.
107
108 :param input_str: The input string.
109 :return: The remaining string and the extracted `re.Match` object, or `None` if the extraction failed.
110 """
111 matched = self._regex.match(input_str)
112 if not matched:
113 return None
114 return input_str[matched.span()[1]:], matched
