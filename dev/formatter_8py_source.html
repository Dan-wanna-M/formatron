79 A Formatter that enforces a format on the string generated by a language model.

82 def __init__(self, extractors: list[Extractor], engine: kbnf.Engine,
83 decode_callback: typing.Callable[[list[int]], str], grammar_str: str):
84 """
85 Initialize the formatter.
86 :param extractors: The matchers to extract data from the generated string.
87 :param engine: The KBNF engine to enforce the format.
88 :param decode_callback: The callback to decode the token IDs to a string.
89 :param grammar_str: The KBNF grammar string.
90 """
91 self._extractors = extractors
92 self._engine = engine
93 self._token_ids = []
94 self._decode_callback = decode_callback
95 self._grammar_str = grammar_str
96 self._captures = {}
97
98 @property
99 def grammar_str(self):
100 """
101 Get the KBNF grammar string.
102 """
103 return self._grammar_str
104
105 def accept_token(self, token_id: int):
106 result = self._engine.try_accept_new_token(token_id)
107 self._token_ids.append(token_id)
108 if result == AcceptTokenResult.Finished:
109 output = self._decode_callback(self._token_ids)
110 self.on_completionon_completion(output)
111 return result
112
113 def accept_bytes(self, _bytes: bytes):
114 self._engine.try_accept_new_bytes(_bytes)
115
116 def compute_allowed_tokens(self) -> None:
117 self._engine.compute_allowed_token_ids()
118
119 def mask_logits(self, logits) -> typing.Any:
120 return self._engine.mask_logits(logits)
121
122 def get_allowed_tokens_since_last_computation(self) -> typing.Sequence[int]:
123 return self._engine.get_allowed_token_ids_from_last_computation()
124
125 def get_tokens_to_finish_since_last_computation(self) -> typing.Sequence[int]:
126 return self._engine.get_token_ids_to_finish_from_last_computation()
127
128 def is_completed(self) -> bool:
129 return self._engine.is_finished()
130
131 def on_completion(self, generated_output: str) -> None:
132 for matcher in self._extractors:
133 generated_output, captured = matcher.extract(generated_output)
134 if matcher.capture_name:
135 if matcher.capture_name in self._captures:
136 self._captures[matcher.capture_name] = [self._captures[matcher.capture_name]]
137 self._captures[matcher.capture_name].append(captured)
138 else:
139 self._captures[matcher.capture_name] = captured
140
141 @property
142 def captures(self) -> dict[str, typing.Any] | None:
143 return self._captures
144
145 def reset(self) -> None:
146 self._captures.clear()
147 self._engine.reset()
148 self._token_ids.clear()
149
150 def __str__(self):
151 return str(self._engine)

160 def __init__(self):
161 """
162 Initialize the formatter builder.

166 self._rules = []

173 def _assert_capture_name_valid(self, capture_name: str):
174 assert capture_name.isidentifier(), (f"capture_name {capture_name}"
175 f" should only contains alphanumeric characters, "
176 f"underscores, and does not start with digits!")
177 assert capture_name not in self._capture_names, f"capture_name {capture_name} is duplicated!"

185 def append_multiline_str(self, lines: str) -> None:
186 """
187 Appends a multiline string to the format, preserving the first line's leading whitespaces
188 and remove any common leading whitespaces from subsequent lines.
189
190 Note that tabs and spaces are both treated as whitespace, but they are not equal:
191 the lines " hello" and "\\thello" are considered to have no common leading whitespace.
192
193 Entirely blank lines are normalized to a newline character.
194 """
195 first = lines.find('\n')
196 self.append_str(lines[:first + 1] + textwrap.dedent(lines[first + 1:]))

198 def append_str(self, string: str) -> None:
199 """
200 Append a string to the format without any post-processing.
201 """
202 state = "normal"
203 last = 0
204
205 def append_literal(end):
206 if last < end:
207 literal = string[last:end]
208 self._main_rule.append(repr(literal))
209 self._extractors.append(LiteralExtractor(literal))
210
211 for i, char in enumerate(string):
212 if char == "$":
213 if state != "escaped":
214 state = "dollar"
215 else:
216 state = "normal"
217 elif state == "dollar":
218 if char == "{":
219 append_literal(i - 1)
220 last = i + 1
221 state = "left_bracket"
222 else:
223 state = "normal"
224 elif state == "left_bracket":
225 if char == "}":
226 state = "normal"
227 self._main_rule.append(string[last:i])
228 self._extractors.append(self._nonterminal_to_extractor[string[last:i]])
229 last = i + 1
230 elif char == "\\":
231 state = "escaped"
232 else:
233 state = "normal"
234 append_literal(len(string))

236 def _create_nonterminal(self, capture_name: typing.Optional[str], name: str) -> str:
237 if capture_name is not None:
238 self._assert_capture_name_valid(capture_name)
239 self._capture_names.add(capture_name)
240 nonterminal = f"__{name}_{capture_name}_{self._instance_id}"
241 else:
242 nonterminal = f"__{name}_{self._counter}_{self._instance_id}"
243 self._counter += 1
244 return nonterminal

246 def choose(self, *extractors: Extractor | str, capture_name: str = None) -> ChoiceExtractor:
247 """
248 Create a choice extractor.
249 :param extractors: The extractors to choose from.
250 :param capture_name: The capture name of the extractor, or `None` if the extractor does not capture.
251 :return: The choice extractor.
252 """
253 new_extractors = []
254 for extractor in extractors:
255 if isinstance(extractor, str):
256 new_extractors.append(LiteralExtractor(extractor))
257 else:
258 new_extractors.append(extractor)
259 return self._add_extractor(capture_name, "choice",
260 lambda nonterminal: ChoiceExtractor(new_extractors, capture_name, nonterminal),
261 lambda nonterminal:
262 f"{nonterminal} ::= {' | '.join([i.kbnf_representation for i in new_extractors])};")

264 def _add_extractor(self, capture_name: str, extractor_type: str,
265 create_extractor: typing.Callable[[str], Extractor],
266 create_rule: typing.Callable[[str], str]):
267 nonterminal = self._create_nonterminal(capture_name, extractor_type)
268 self._nonterminal_to_extractor[nonterminal] = create_extractor(nonterminal)
269 self._rules.append(create_rule(nonterminal))
270 return self._nonterminal_to_extractor[nonterminal]

299 def str(self, *, stop: typing.Union[str, list[str]] = None,
300 not_contain: typing.Union[str, list[str], None] = None,
301 capture_name: typing.Optional[str] = None) -> RegexExtractor:
302 """
303 Create a string extractor.
304 :param stop: The strings for the extractors to stop at. They will be included in text generation and extraction.
305 :param not_contain: The strings that should not be included in the generation.
306 They will not be included in the generation and extraction.
307 :param capture_name: The capture name of the extractor, or `None` if the extractor does not capture.
308 :return: The string extractor.
309 """
310 stop = [stop] if isinstance(stop, str) else stop or []
311 not_contain = [not_contain] if isinstance(not_contain, str) else not_contain or []
312 nonterminal = self._create_nonterminal(capture_name, "str")
313 if not stop and not not_contain:
314 capture_regex = ".*"
315 nonterminal_regex = "#'.*'"
316 else:
317 capture_regex = f".*?(?:{'|'.join(map(repr, stop + not_contain))})"
318 excepted = f"{nonterminal}_excepted"
319 end = f"({'|'.join(map(repr, stop))})" if stop else ""
320 nonterminal_regex = f"except!({excepted}){end}"
321 self._rules.append(f"{excepted} ::= {' | '.join(map(repr, stop + not_contain))};")
322 self._rules.append(f"{nonterminal} ::= {nonterminal_regex};")
323 self._nonterminal_to_extractor[nonterminal] = RegexExtractor(capture_regex, capture_name, nonterminal)
324 return self._nonterminal_to_extractor[nonterminal]

326 def build(self, vocabulary: kbnf.Vocabulary,
327 decode: typing.Callable[[list[int]], str],
328 engine_config: kbnf.Config = None) -> Formatter:
329 """
330 Build a formatter from the builder. The builder will not be consumed and can be used again.
331 :param vocabulary: The KBNF engine vocabulary for the formatter.
332 :param decode: The callback to decode the token IDs to a string.
333 :param engine_config: The KBNF engine configuration.
334 :return: The formatter.
335 """
336 assert len(self._main_rule) != 0, "An empty formatter builder cannot build!"
337 rules = copy(self._rules)
338 rules.append(f"start ::= {' '.join(self._main_rule)};")
339 grammar_str = "\n".join(rules)
340 engine = Engine(grammar_str, vocabulary, engine_config)
341 extractors = copy(self._extractors)
342 f = Formatter(extractors, engine, decode, grammar_str)
343 return f
