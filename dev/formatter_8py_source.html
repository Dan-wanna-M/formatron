93class Formatter(FormatterBase):
94 """
95 A Formatter that enforces a format on the string generated by a language model. It is designed to compose
96 multiple extractors in a sequential, unambiguous, greedy manner. Check out the Formatter.captures property docs for more details.
97 If you need more complex extraction logic, you need to implement your own Extractor.
98 """
99
100 def __init__(self, extractors: list[Extractor], engine: kbnf.Engine,
101 decode_callback: typing.Callable[[list[int]], str], grammar_str: str):
102 """
103 Initialize the formatter.
104 Args:
105 extractors: The matchers to extract data from the generated string.
106 engine: The KBNF engine to enforce the format.
107 decode_callback: The callback to decode the token IDs to a string.
108 grammar_str: The KBNF grammar string.
109 """
110 self._extractors = extractors
111 self._engine = engine
112 self._token_ids = []
113 self._decode_callback = decode_callback
114 self._grammar_str = grammar_str
115 self._captures = {}
116
117 @property
118 def grammar_str(self):
119 """
120 Get the KBNF grammar string.
121 """
122 return self._grammar_str
123
124 def accept_token(self, token_id: int):
125 result = self._engine.try_accept_new_token(token_id)
126 self._token_ids.append(token_id)
127 if result == kbnf.AcceptTokenResult.Finished:
128 output = self._decode_callback(self._token_ids)
129 self._on_completion_on_completion(output)
130 return result
131
132 def accept_bytes(self, _bytes: bytes):
133 self._engine.try_accept_new_bytes(_bytes)
134
135 def compute_allowed_tokens(self) -> None:
136 self._engine.compute_allowed_token_ids()
137
138 def mask_logits(self, logits) -> typing.Any:
139 return self._engine.mask_logits(logits)
140
141 def get_allowed_tokens_since_last_computation(self) -> typing.Sequence[int]:
142 return self._engine.get_allowed_token_ids_from_last_computation()
143
144 def is_completed(self) -> bool:
145 """
146 Check if the generation is completed. This means the generation is ended by the engine.
147 If the generation is ended by integration-specific stop conditions like `max_new_tokens`,
148 the generation is not considered completed by this method.
149 """
150 return self._engine.is_finished()
151
152 def _on_completion(self, generated_output: str) -> None:
153 for matcher in self._extractors:
154 result = matcher.extract(generated_output)
155 if result is None:
156 captured = None
157 else:
158 generated_output, captured = matcher.extract(generated_output)
159 if matcher.capture_name:
160 if matcher.capture_name in self._captures:
161 self._captures[matcher.capture_name] = [
162 self._captures[matcher.capture_name]]
163 self._captures[matcher.capture_name].append(captured)
164 else:
165 self._captures[matcher.capture_name] = captured
166
167 @property
168 def captures(self) -> dict[str, typing.Any] | None:
169 """
170 Get the captures from the generated string. Note that the captures are only available for one extractor if:
171 - The extractor has a capture name.
172 - Formatter.is_completed() returns True.
173 - The extractor successfully extracts the data.
174 - This means the extractor identifies the correct string span to extract and whatever post-processing the extractor does on the extracted string is successful.
175
176 Captures are obtained by calling `Extractor.extract` method on the generated string in the sequence of extractors appended to the formatter.
177 Note that the previous extractors does not 'see' the semantics of the later extractors. For example,
178 consider the following formatter:
179 ```python
180 f = FormatterBuilder()
181 f.append_line(f"{f.regex('.*?', capture_name='a')}{f.regex('.*', capture_name='b')}")
182 f = f.build()
183 ```
184 The `b` extractor will always corresponding to `None` because the `a` extractor will always extract the whole string.
185 This behavior is different from what a typical regular expression engine would do!
186 """
187 return self._captures
188
189 def reset(self) -> None:
190 self._captures.clear()
191 self._engine.reset()
192 self._token_ids.clear()
193
194 def __str__(self):
195 return (f"Formatter(engine={self._engine}, "
196 f"captures={self._captures}, "
197 f"extractors={len(self._extractors)}, "
198 f"completed={self.is_completed()}, "
199 f"token_ids={len(self._token_ids)})"
200 f"grammar={self._grammar_str})")

223 def _assert_capture_name_valid(self, capture_name: str):
224 assert capture_name.isidentifier(), (f"capture_name {capture_name}"

229 def append_line(self, line: str) -> None:
230 """
231 Append a line to the format. Specifically, a newline character is appended to the input.
232
233 Note that if you need a literal `$`, you need to escape it by adding a backslash: `\\$`.
234 """
235 self.append_str(line + '\n')
236
237 def append_multiline_str(self, lines: str) -> None:
238 """
239 Appends a multiline string to the format, preserving the first line's leading whitespaces
240 and remove any common leading whitespaces from subsequent lines.
241
242 Note that tabs and spaces are both treated as whitespace, but they are not equal:
243 the lines " hello" and "\\thello" are considered to have no common leading whitespace.
244
245 Entirely blank lines are normalized to a newline character.
246
247 Note that if you need a literal `$`, you need to escape it by adding a backslash: `\\$`.
248 """
249 first = lines.find('\n')
250 self.append_str(lines[:first + 1] + textwrap.dedent(lines[first + 1:]))

252 def append_str(self, string: str) -> None:
253 """
254 Append a string to the format without any post-processing.
255
256 Note that if you need a literal `$`, you need to escape it by adding a backslash: `\\$`.
257 """
258 state = "normal"
259 last = 0
260
261 def append_literal(end):
262 if last < end:
263 literal = string[last:end]
264 self._main_rule.append(repr(literal))
265 self._extractors.append(LiteralExtractor(literal))
266
267 for i, char in enumerate(string):
268 if char == "$":
269 if state != "escaped":
270 state = "dollar"
271 else:
272 state = "normal"
273 elif state == "dollar":
274 if char == "{":
275 append_literal(i - 1)
276 last = i + 1
277 state = "left_bracket"
278 else:
279 state = "normal"
280 elif state == "left_bracket":
281 if char == "}":
282 state = "normal"
283 self._main_rule.append(string[last:i])
284 self._extractors.append(
285 self._nonterminal_to_extractor[string[last:i]])
286 last = i + 1
287 elif char == "\\":
288 state = "escaped"
289 else:
290 state = "normal"
291 append_literal(len(string))

304 def choose(self, *extractors: Extractor | str, capture_name: str = None) -> ChoiceExtractor:
305 """
306 Create a choice extractor.
307
308 Check out the ChoiceExtractor docs for more details.
309 Args:
310 extractors: The extractors to choose from.
311 capture_name: The capture name of the extractor, or `None` if the extractor does not capture.
312 Returns:
313 The choice extractor.
314 """
315 new_extractors = []
316 for extractor in extractors:
317 if isinstance(extractor, str):
318 new_extractors.append(LiteralExtractor(extractor))
319 else:
320 new_extractors.append(extractor)
321 return self._add_extractor("choice",
322 lambda nonterminal: ChoiceExtractor(new_extractors, capture_name, nonterminal))

324 def _add_extractor(self, extractor_type: str, create_extractor: typing.Callable[[str], Extractor]):
325 nonterminal = self._create_nonterminal(extractor_type)
326 extractor = create_extractor(nonterminal)
327 if isinstance(extractor, NonterminalExtractor):
328 self._add_capture_name(extractor)
329 nonterminal = extractor.nonterminal
330 self._nonterminal_to_extractor[nonterminal] = extractor
331 self._rules.append(extractor.kbnf_definition)
332 return extractor

345 """

383 def str(self, *, stop: typing.Union[str, list[str]] = None,
384 capture_name: typing.Optional[str] = None) -> Extractor:
385 """
386 Create a string extractor.
387
388 The extractor will extract all text until(inclusive) one of the stop strings is encountered.
389
390 Args:
391 stop: The strings for the extractors to stop at. They will be included in text generation and extraction.
392 capture_name: The capture name of the extractor, or `None` if the extractor does not capture.
393 Returns:
394 The string extractor.
395 """
396 stop = [stop] if isinstance(stop, str) else stop or []
397 if not stop:
398 capture_regex = ".*"
399 else:
400 backslash = '\\'
401 capture_regex = f".*?(?:{'|'.join([i.replace(backslash, backslash * 2) for i in map(re.escape, stop)])})"
402 print(f"{capture_regex}")
403 return self._add_extractor("str",
404 lambda nonterminal: RegexExtractor(capture_regex, capture_name, nonterminal))

423 def build(self, vocabulary: kbnf.Vocabulary,
424 decode: typing.Callable[[list[int]], str],
425 engine_config: kbnf.Config = None) -> Formatter:
426 """
427 Build a formatter from the builder. The builder will not be consumed and can be used again.
428
429 Args:
430 vocabulary: The KBNF engine vocabulary for the formatter.
431 decode: The callback to decode the token IDs to a string.
432 engine_config: The KBNF engine configuration.
433 Returns:
434 The formatter.
435 """
436 assert len(
437 self._main_rule) != 0, "An empty formatter builder cannot build!"
438 rules = copy(self._rules)
439 rules.append(f"start ::= {' '.join(self._main_rule)};")
440 grammar_str = "\n".join(rules)
441 engine = kbnf.Engine(grammar_str, vocabulary, engine_config)
442 extractors = copy(self._extractors)
443 f = Formatter(extractors, engine, decode, grammar_str)
444 return f
