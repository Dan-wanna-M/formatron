87 """
88 A Formatter that enforces a format on the string generated by a language model.

91 def __init__(self, extractors: list[Extractor], engine: kbnf.Engine,
92 decode_callback: typing.Callable[[list[int]], str], grammar_str: str):
93 """
94 Initialize the formatter.
95 :param extractors: The matchers to extract data from the generated string.
96 :param engine: The KBNF engine to enforce the format.
97 :param decode_callback: The callback to decode the token IDs to a string.
98 :param grammar_str: The KBNF grammar string.
99 """
100 self._extractors = extractors
101 self._engine = engine
102 self._token_ids = []
103 self._decode_callback = decode_callback
104 self._grammar_str = grammar_str
105 self._captures = {}
106
107 @property
108 def grammar_str(self):
109 """
110 Get the KBNF grammar string.
111 """
112 return self._grammar_str
113
114 def accept_token(self, token_id: int):
115 result = self._engine.try_accept_new_token(token_id)
116 self._token_ids.append(token_id)
117 if result == AcceptTokenResult.Finished:
118 output = self._decode_callback(self._token_ids)
119 self.on_completionon_completion(output)
120 return result
121
122 def accept_bytes(self, _bytes: bytes):
123 self._engine.try_accept_new_bytes(_bytes)
124
125 def compute_allowed_tokens(self) -> None:
126 self._engine.compute_allowed_token_ids()
127
128 def mask_logits(self, logits) -> typing.Any:
129 return self._engine.mask_logits(logits)
130
131 def get_allowed_tokens_since_last_computation(self) -> typing.Sequence[int]:
132 return self._engine.get_allowed_token_ids_from_last_computation()
133
134 def is_completed(self) -> bool:
135 return self._engine.is_finished()
136
137 def on_completion(self, generated_output: str) -> None:
138 for matcher in self._extractors:
139 result = matcher.extract(generated_output)
140 if result is None:
141 captured = None
142 else:
143 generated_output, captured = matcher.extract(generated_output)
144 if matcher.capture_name:
145 if matcher.capture_name in self._captures:
146 self._captures[matcher.capture_name] = [self._captures[matcher.capture_name]]
147 self._captures[matcher.capture_name].append(captured)
148 else:
149 self._captures[matcher.capture_name] = captured
150
151 @property
152 def captures(self) -> dict[str, typing.Any] | None:
153 return self._captures
154
155 def reset(self) -> None:
156 self._captures.clear()
157 self._engine.reset()
158 self._token_ids.clear()
159
160 def __str__(self):
161 return str(self._engine)

166 A builder for creating a Formatter.

170 def __init__(self):
171 """
172 Initialize the formatter builder.
173 """
174 self._counter = 0
175 self._main_rule = []
176 self._rules = []
177 self._capture_names = set()
178 self._nonterminal_to_extractor = {}
179 self._extractors = []
180 self._instance_id = self.__class__._formatter_builder_counter
181 self.__class__._formatter_builder_counter += 1

183 def _assert_capture_name_valid(self, capture_name: str):
184 assert capture_name.isidentifier(), (f"capture_name {capture_name}"
185 f" should only contains alphanumeric characters, "
186 f"underscores, and does not start with digits!")
187 assert capture_name not in self._capture_names, f"capture_name {capture_name} is duplicated!"

189 def append_line(self, line: str) -> None:
190 """
191 Append a line to the format. Specifically, a newline character is appended to the input.
192 """
193 self.append_str(line + '\n')

195 def append_multiline_str(self, lines: str) -> None:
196 """
197 Appends a multiline string to the format, preserving the first line's leading whitespaces
198 and remove any common leading whitespaces from subsequent lines.
199
200 Note that tabs and spaces are both treated as whitespace, but they are not equal:
201 the lines " hello" and "\\thello" are considered to have no common leading whitespace.
202
203 Entirely blank lines are normalized to a newline character.
204 """
205 first = lines.find('\n')
206 self.append_str(lines[:first + 1] + textwrap.dedent(lines[first + 1:]))
207
208 def append_str(self, string: str) -> None:
209 """
210 Append a string to the format without any post-processing.
211 """
212 state = "normal"
213 last = 0
214
215 def append_literal(end):
216 if last < end:
217 literal = string[last:end]
218 self._main_rule.append(repr(literal))
219 self._extractors.append(LiteralExtractor(literal))
220
221 for i, char in enumerate(string):
222 if char == "$":
223 if state != "escaped":
224 state = "dollar"
225 else:
226 state = "normal"
227 elif state == "dollar":
228 if char == "{":
229 append_literal(i - 1)
230 last = i + 1
231 state = "left_bracket"
232 else:
233 state = "normal"
234 elif state == "left_bracket":
235 if char == "}":
236 state = "normal"
237 self._main_rule.append(string[last:i])
238 self._extractors.append(self._nonterminal_to_extractor[string[last:i]])
239 last = i + 1
240 elif char == "\\":
241 state = "escaped"
242 else:
243 state = "normal"
244 append_literal(len(string))
245
246 def _create_nonterminal(self, capture_name: typing.Optional[str], name: str) -> str:
247 if capture_name is not None:
248 self._assert_capture_name_valid(capture_name)
249 self._capture_names.add(capture_name)
250 nonterminal = f"__{name}_{capture_name}_{self._instance_id}"
251 else:
252 nonterminal = f"__{name}_{self._counter}_{self._instance_id}"
253 self._counter += 1
254 return nonterminal

256 def choose(self, *extractors: Extractor | str, capture_name: str = None) -> ChoiceExtractor:
257 """
258 Create a choice extractor.
259 :param extractors: The extractors to choose from.
260 :param capture_name: The capture name of the extractor, or `None` if the extractor does not capture.
261 :return: The choice extractor.
262 """
263 new_extractors = []
264 for extractor in extractors:
265 if isinstance(extractor, str):
266 new_extractors.append(LiteralExtractor(extractor))
267 else:
268 new_extractors.append(extractor)
269 return self._add_extractor(capture_name, "choice",
270 lambda nonterminal: ChoiceExtractor(new_extractors, capture_name, nonterminal),
271 lambda nonterminal:
272 f"{nonterminal} ::= {' | '.join([i.kbnf_representation for i in new_extractors])};")

274 def _add_extractor(self, capture_name: str, extractor_type: str,
275 create_extractor: typing.Callable[[str], Extractor],
276 create_rules: typing.Callable[[str], str]):
277 nonterminal = self._create_nonterminal(capture_name, extractor_type)
278 self._nonterminal_to_extractor[nonterminal] = create_extractor(nonterminal)
279 self._rules.append(create_rules(nonterminal))
280 return self._nonterminal_to_extractor[nonterminal]

282 def extractor(self, create_extractor: typing.Callable[[str], Extractor],
283 create_rules: typing.Callable[[str], str], capture_name: str = None) -> Extractor:
284 """
285 Create a custom extractor.
286 :param create_extractor: callable with signature (extractor_nonterminal: str)->Extractor that create the extractor. extractor_nonterminal is the auto-generated nonterminal reference for the extractor.
287 :param create_rules: callable with signature (extractor_nonterminal: str)->str that create the KBNF rules for the extractor. It is separated from create_extractor to allow more flexibility. For example, you can reuse the same extractor with different rules.
288 :param capture_name: The capture name of the extractor, or `None` if the extractor does not capture.

327 def str(self, *, stop: typing.Union[str, list[str]] = None,
328 capture_name: typing.Optional[str] = None) -> RegexExtractor:
329 """
330 Create a string extractor.
331 :param stop: The strings for the extractors to stop at. They will be included in text generation and extraction.
332 :param capture_name: The capture name of the extractor, or `None` if the extractor does not capture.
333 :return: The string extractor.
334 """
335 stop = [stop] if isinstance(stop, str) else stop or []
336 nonterminal = self._create_nonterminal(capture_name, "str")
337 if not stop:
338 capture_regex = ".*"
339 nonterminal_regex = "#'.*'"
340 else:
341 backslash = '\\'
342 capture_regex = f".*?(?:{'|'.join([i.replace(backslash, backslash * 2) for i in map(re.escape, stop)])})"
343 nonterminal_regex = f"#e'{capture_regex}'"
344 self._rules.append(f"{nonterminal} ::= {nonterminal_regex};")
345 self._nonterminal_to_extractor[nonterminal] = RegexExtractor(capture_regex, capture_name, nonterminal)
346 return self._nonterminal_to_extractor[nonterminal]

348 def build(self, vocabulary: kbnf.Vocabulary,
349 decode: typing.Callable[[list[int]], str],
350 engine_config: kbnf.Config = None) -> Formatter:
351 """
352 Build a formatter from the builder. The builder will not be consumed and can be used again.
353 :param vocabulary: The KBNF engine vocabulary for the formatter.
354 :param decode: The callback to decode the token IDs to a string.
355 :param engine_config: The KBNF engine configuration.
356 :return: The formatter.
357 """
358 assert len(self._main_rule) != 0, "An empty formatter builder cannot build!"
359 rules = copy(self._rules)
360 rules.append(f"start ::= {' '.join(self._main_rule)};")
361 grammar_str = "\n".join(rules)
362 engine = Engine(grammar_str, vocabulary, engine_config)
363 extractors = copy(self._extractors)
364 f = Formatter(extractors, engine, decode, grammar_str)
365 return f
