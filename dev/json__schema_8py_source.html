148def _obtain_type(schema: dict[str, typing.Any], json_schema_id_to_schema:dict[int, typing.Type]) -> typing.Type[typing.Any|None]:
149 """
150 Directly obtain type information from this schema level.
151 """
152
153 if "type" not in schema:
154 obtained_type = None
155 else:
156 json_type = schema["type"]
157 if json_type == "string":
158 obtained_type = str
159 elif json_type == "number":
160 obtained_type = float
161 elif json_type == "integer":
162 obtained_type = int
163 elif json_type == "boolean":
164 obtained_type = bool
165 elif json_type == "null":
166 obtained_type = type(None)
167 elif json_type == "array":
168 obtained_type = typing.List
169 elif json_type == "object":
170 obtained_type = object
171 elif isinstance(json_type, collections.abc.Sequence):
172 new_list = []
173 for item in json_type:
174 new_schema = schema.copy()
175 new_schema["type"] = item
176 new_list.append(_obtain_type(new_schema, json_schema_id_to_schema))
177 obtained_type = typing.Union[tuple(new_list)]
178 else:
179 raise TypeError(f"Unsupported type in json schema: {json_type}")
180 literal = _get_literal(schema)
181 if literal is not None:
182 return _handle_literal(literal, obtained_type, schema, json_schema_id_to_schema)
183 return obtained_type

189def _merge_referenced_schema(schema: dict[str, typing.Any], memo: set[int]):
190 keys = ["$ref", "$dynamicRef"]
191 if id(schema) in memo: # Circular reference
192 return None
193 memo.add(id(schema))
194 if isinstance(schema, list):
195 for item in schema:
196 _merge_referenced_schema(item, memo)
197 elif isinstance(schema, dict):
198 for key in keys:
199 if key in schema:
200 _merge_referenced_schema(schema[key], memo) # ensure no unmerged references
201 for ref_key, ref_value in schema[key].items():
202 _merge_key(schema, ref_key, ref_value)
203 schema[ref_key] = ref_value
204 del schema[key]
205 for key, value in schema.items():
206 if key not in keys:
207 _merge_referenced_schema(value, memo)

209def _merge_key(schema:dict[str, typing.Any], ref_key:str, reference_value:typing.Any):
210 if ref_key not in schema:
211 schema[ref_key] = reference_value
212 return None
213 if isinstance(schema[ref_key], dict) and isinstance(reference_value, dict):
214 for new_ref_key, new_ref_value in reference_value.items():
215 _merge_key(schema[ref_key], new_ref_key, new_ref_value)
216 return None
217 raise ValueError(f"Duplicate keys in schema referenced by {ref_key} in JSON schema: {schema} is not supported")

220def _recursive_resolve_reference(base_uri: str, schema: typing.Any, registry: Registry):
221 if isinstance(schema, list):
222 new_list = []
223 for item in schema:
224 new_list.append(_recursive_resolve_reference(base_uri, item, registry))
225 schema.clear()
226 schema.extend(new_list)
227 if isinstance(schema, dict):
228 if "$id" in schema:
229 base_uri = _resolve_new_url(base_uri, schema["$id"])
230 resolver = registry.resolver(base_uri)
231 keys = ["$ref", "$dynamicRef"]
232 for key in keys:
233 if key in schema:
234 _resolve_reference(schema, key, resolver)
235 for key, value in schema.items():
236 if key not in keys:
237 _recursive_resolve_reference(base_uri, value, registry)
238 return schema

240def _resolve_reference(schema: dict[str, typing.Any], key: str, resolver: typing.Any):

242 if resolved.contents is schema:

244 schema[key] = resolved.contents
