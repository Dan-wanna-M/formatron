15def create_engine_vocabulary(tokenizer: ExLlamaV2Tokenizer) -> kbnf.Vocabulary:
16 """
17 Create a vocabulary for the KBNF engine.
18 """
19 assert hasattr(tokenizer.tokenizer_model, "vocab"), (f"tokenizer({tokenizer})"
20 f" with tokenizer_model({tokenizer.tokenizer_model})"
21 f" does not have vocab attribute!")
22 vocab = {tokenizer.tokenizer_model.id_to_piece(i): i for i in range(tokenizer.tokenizer_model.vocab_size())}
23 new_vocab = get_original_characters(vocab)
24 return kbnf.Vocabulary({v: kbnf.Token(k) for k, v in new_vocab.items()},
25 {k: v for k, v in enumerate(vocab)})

28def create_formatter_filter(model: ExLlamaV2, tokenizer: ExLlamaV2Tokenizer,
29 formatter_builder: FormatterBuilder,

31 """
32 Create a formatter filter for the ExLlamaV2 engine.
33 """
34 vocab = create_engine_vocabulary(tokenizer)
35 f = formatter_builder.build(vocab, lambda tokens: tokenizer.decode(torch.tensor(tokens)))
