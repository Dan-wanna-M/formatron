14def create_engine_vocabulary(tokenizer: PreTrainedTokenizerBase) -> kbnf.Vocabulary:
15 """
16 Create a vocabulary for the KBNF engine.
17 """
18 vocab = tokenizer.get_vocab()
19 new_vocab = get_original_characters(vocab)
20 return kbnf.Vocabulary({v: kbnf.Token(k) for k, v in new_vocab.items()},
21 {v: k for k, v in vocab.items()})
